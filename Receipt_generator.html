    <script>

        document.addEventListener('DOMContentLoaded', () => {

            const form = document.getElementById('receipt-form');

            const lineItemsContainer = document.getElementById('line-items-container');

            const addItemBtn = document.getElementById('add-item-btn');

            const printBtn = document.getElementById('print-btn');

            // --- Helper Functions ---

            // Format date to "Month Day, Year"

            const formatDate = (dateString) => {

                if (!dateString) return '';

                const date = new Date(dateString + 'T00:00:00'); // Fix for time-zone issues

                return date.toLocaleDateString('en-US', {

                    year: 'numeric',

                    month: 'long',

                    day: 'numeric',

                });

            };

            // Format number to currency

            const formatCurrency = (amount) => {

                return new Intl.NumberFormat('en-US', {

                    style: 'currency',

                    currency: 'USD',

                }).format(amount);

            };

            // --- Core Functions ---

            // Create a new line item input row

            const createLineItem = () => {

                const itemRow = document.createElement('div');

                itemRow.className = 'grid grid-cols-12 gap-x-2 gap-y-3 line-item-row items-end';

               

                itemRow.innerHTML = `

                    <div class="col-span-12 sm:col-span-6">

                        <label class="block text-sm font-medium text-gray-700">Service Description</label>

                        <input type="text" name="description" placeholder="Math Tutoring" class="mt-1 block w-full rounded-md border-gray-300 shadow-sm focus:border-indigo-500 focus:ring-indigo-500">

                    </div>

                    <div class="col-span-6 sm:col-span-2">

                        <label class="block text-sm font-medium text-gray-700">Qty/Hrs</label>

                        <input type="number" name="quantity" value="1" min="0" step="0.1" class="mt-1 block w-full rounded-md border-gray-300 shadow-sm focus:border-indigo-500 focus:ring-indigo-500">

                    </div>

                    <div class="col-span-6 sm:col-span-2">

                        <label class="block text-sm font-medium text-gray-700">Rate ($)</label>

                        <input type="number" name="rate" value="50" min="0" step="0.01" class="mt-1 block w-full rounded-md border-gray-300 shadow-sm focus:border-indigo-500 focus:ring-indigo-500">

                    </div>

                    <div class="col-span-12 sm:col-span-2 text-right">

                        <button type="button" class="remove-item-btn px-3 py-2 border border-transparent text-sm font-medium rounded-md text-white bg-red-600 hover:bg-red-700 focus:outline-none focus:ring-2 focus:ring-offset-2 focus:ring-red-500">

                            Remove

                        </button>

                    </div>

                `;

                // Add event listener for the remove button

                itemRow.querySelector('.remove-item-btn').addEventListener('click', () => {

                    itemRow.remove();

                    updatePreview();

                });

               

                return itemRow;

            };

            // Update the receipt preview based on form inputs

            const updatePreview = () => {

                // Business & Client Info

                document.getElementById('preview-business-name').textContent = document.getElementById('business-name').value || 'Your Business Name';

                const businessContact = document.getElementById('business-contact').value || 'Your Contact';

                document.getElementById('preview-business-contact').textContent = businessContact;

                document.getElementById('footer-business-contact').textContent = businessContact;

               

                document.getElementById('preview-client-name').textContent = document.getElementById('client-name').value || 'Client Name';

                document.getElementById('preview-student-name').textContent = 'Student: ' + (document.getElementById('student-name').value || 'Student Name');

               

                // Receipt Info

                document.getElementById('preview-receipt-number').textContent = document.getElementById('receipt-number').value || '0001';

                document.getElementById('preview-date').textContent = formatDate(document.getElementById('receipt-date').value);

                document.getElementById('preview-payment-method').textContent = document.getElementById('payment-method').value;

                // Line Items & Calculations

                const itemsBody = document.getElementById('preview-items-body');

                itemsBody.innerHTML = ''; // Clear existing items

               

                let subtotal = 0;

                const itemRows = lineItemsContainer.querySelectorAll('.line-item-row');

               

                itemRows.forEach(row => {

                    const description = row.querySelector('input[name="description"]').value || 'Service';

                    const quantity = parseFloat(row.querySelector('input[name="quantity"]').value) || 0;

                    const rate = parseFloat(row.querySelector('input[name="rate"]').value) || 0;

                    const amount = quantity * rate;

                    subtotal += amount;

                   

                    const previewRow = document.createElement('tr');

                    previewRow.className = 'border-b border-gray-100';

                    previewRow.innerHTML = `

                        <td class="p-3 align-top">${description}</td>

                        <td class="p-3 text-center align-top">${quantity}</td>

                        <td class="p-3 text-right align-top">${formatCurrency(rate)}</td>

                        <td class="p-3 text-right align-top font-medium">${formatCurrency(amount)}</td>

                    `;

                    itemsBody.appendChild(previewRow);

                });

                // Summary Calculations

                const discount = parseFloat(document.getElementById('discount').value) || 0;

                const taxRate = parseFloat(document.getElementById('tax').value) || 0;

                const taxAmount = (subtotal - discount) * (taxRate / 100);

                const total = subtotal - discount + taxAmount;

                // Update Summary Preview

                document.getElementById('preview-subtotal').textContent = formatCurrency(subtotal);

                document.getElementById('preview-discount').textContent = `-${formatCurrency(discount)}`;

                document.getElementById('preview-tax-rate').textContent = taxRate;

                document.getElementById('preview-tax').textContent = formatCurrency(taxAmount);

                document.getElementById('preview-total').textContent = formatCurrency(total);

                document.getElementById('preview-total-paid').textContent = formatCurrency(total);

            };

            // --- Event Listeners ---

           

            // Add new line item

            addItemBtn.addEventListener('click', () => {

                lineItemsContainer.appendChild(createLineItem());
